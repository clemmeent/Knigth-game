240    /**
241     * ProcÃ©dure qui controle l'action sur les boutons ou la zone de texte

245    public void actionPerformed (final ActionEvent pE) 
246    {
247        if (pE.getSource() == this.aButtonNorth){
248            this.aEngine.interpretCommand("go north");
249        }else if (pE.getSource() == this.aButtonSouth){
250            this.aEngine.interpretCommand("go south");
251        }else if (pE.getSource() == this.aButtonEast){
252            this.aEngine.interpretCommand("go east");
253        }else if (pE.getSource() == this.aButtonWest){
254            this.aEngine.interpretCommand("go west");
255        }else if (pE.getSource() == this.aButtonUp){
256            this.aEngine.interpretCommand("go up");
257        }else if (pE.getSource() == this.aButtonDown){
258            this.aEngine.interpretCommand("go down");
259        }else if (pE.getSource() == this.aButtonQuit){
260            this.aEngine.interpretCommand("quit");
261        }else if (pE.getSource() == this.aButtonHelp){
262            this.aEngine.interpretCommand("help");
263        }else if (pE.getSource() == this.aButtonBack){
264            this.aEngine.interpretCommand("back");
265        }else if (pE.getSource() == this.aButtonInventory){
266            this.aEngine.interpretCommand("inventory");
267        }else{
268            this.processCommand();
269        }// else
270    }// actionPerformed ()
