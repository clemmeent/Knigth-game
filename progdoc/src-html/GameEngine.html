170    /**
171     * ProcÃ©dure qui transforme et analyse la commande de l'utilisateur
172     * 
173     * @param pStringCommand Chaine de caractÃ¨re tapÃ©e au clavier
174     */
175    public void interpretCommand (final String pStringCommand) 
176    {
177        this.aGui.println("\n" + "> " + pStringCommand ); //Affiche les caractÃ¨res tapÃ©s au clavier
178        Command vCommand = this.aParser.getCommand(pStringCommand); // Converti en objet de type Command
179        
180        if (vCommand.isUnknown()){
181            this.aGui.println("I don't know what you mean...");
182        }else{
183            String vCommandWord = vCommand.getCommandWord();
184            if (vCommandWord.equals("go")){
185                this.goRoom(vCommand);
186            }else if (vCommandWord.equals("look")){
187                this.look(vCommand);
188            }else if (vCommandWord.equals("eat")){
189                this.eat(vCommand);
190            }else if (vCommandWord.equals("quit")){
191                if ( vCommand.hasSecondWord() ){
192                    this.aGui.println( "Quit what?" );
193                }else{
194                    this.endGame();
195                }// else
196            }else if (vCommandWord.equals("help")){
197                this.printHelp();
198            }else if (vCommandWord.equals("back")){
199                this.back(vCommand);
200            }else if (vCommandWord.equals("test")){
201                this.test(vCommand);
202            }else if (vCommandWord.equals("take")){
203                this.take(vCommand);
204            }else if (vCommandWord.equals("drop")){
205                this.drop(vCommand);
206            }else if (vCommandWord.equals("inventory")){
207                this.inventory(vCommand);
208            }else if (vCommandWord.equals("charge")){
209                this.charge(vCommand);
210            }else if (vCommandWord.equals("fire")){
211                this.fire(vCommand);
212            }else{
213                this.aGui.println("Unknown command!");
214            }// else
215        }// else
216    }// interpretCommand ()
